1   package eu.fbk.rdfpro.rules.functions;
2   
3   import org.openrdf.model.Value;
4   import org.openrdf.model.ValueFactory;
5   import org.openrdf.model.vocabulary.RDF;
6   import org.openrdf.query.algebra.evaluation.ValueExprEvaluationException;
7   import org.openrdf.query.algebra.evaluation.function.Function;
8   
9   import eu.fbk.rdfpro.rules.RR;
10  
11  public class StarSelectGraphFunction implements Function {
12  
13      @Override
14      public String getURI() {
15          return RR.STAR_SELECT_GRAPH.stringValue();
16      }
17  
18      @Override
19      public Value evaluate(final ValueFactory valueFactory, final Value... args)
20              throws ValueExprEvaluationException {
21  
22          final Value global = args[0];
23  
24          Value result = null;
25          for (int i = 1; i < args.length; ++i) {
26              final Value graph = args[i];
27              if (!graph.equals(global)) {
28                  if (result == null) {
29                      result = graph;
30                  } else {
31                      return RDF.NIL;
32                  }
33              }
34          }
35  
36          return result != null ? result : global;
37      }
38  
39  }
