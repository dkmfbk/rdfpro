1   package eu.fbk.rdfpro.rules.model;
2   
3   import java.util.HashSet;
4   import java.util.Iterator;
5   import java.util.Objects;
6   import java.util.Set;
7   
8   import javax.annotation.Nullable;
9   
10  import org.openrdf.model.Namespace;
11  import org.openrdf.model.Resource;
12  import org.openrdf.model.Statement;
13  import org.openrdf.model.URI;
14  import org.openrdf.model.Value;
15  import org.openrdf.model.impl.NamespaceImpl;
16  import org.openrdf.model.util.ModelException;
17  import org.openrdf.repository.RepositoryConnection;
18  import org.openrdf.repository.RepositoryException;
19  import org.openrdf.repository.RepositoryResult;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  final class RepositoryQuadModel extends QuadModel {
24  
25      private static final Logger LOGGER = LoggerFactory.getLogger(RepositoryQuadModel.class);
26  
27      private static final long serialVersionUID = 1L;
28  
29      private final RepositoryConnection connection;
30  
31      public RepositoryQuadModel(final RepositoryConnection connection) {
32          this.connection = Objects.requireNonNull(connection);
33      }
34  
35      @Override
36      protected Set<Namespace> doGetNamespaces() {
37          try {
38              final Set<Namespace> namespaces = new HashSet<>();
39              RepositoryResult<? extends Namespace> iteration;
40              iteration = this.connection.getNamespaces();
41              try {
42                  while (iteration.hasNext()) {
43                      namespaces.add(iteration.next());
44                  }
45              } finally {
46                  iteration.close();
47              }
48              return namespaces;
49          } catch (final RepositoryException ex) {
50              throw new ModelException(ex);
51          }
52      }
53  
54      @Override
55      @Nullable
56      protected Namespace doGetNamespace(final String prefix) {
57          try {
58              final String name = this.connection.getNamespace(prefix);
59              return name == null ? null : new NamespaceImpl(prefix, name);
60          } catch (final RepositoryException ex) {
61              throw new ModelException(ex);
62          }
63      }
64  
65      @Override
66      protected Namespace doSetNamespace(final String prefix, @Nullable final String name) {
67          try {
68              final String oldName = this.connection.getNamespace(prefix);
69              final Namespace oldNamespace = oldName == null ? null : new NamespaceImpl(prefix,
70                      oldName);
71              if (name == null) {
72                  this.connection.removeNamespace(prefix);
73              } else {
74                  this.connection.setNamespace(prefix, name);
75              }
76              return oldNamespace;
77          } catch (final RepositoryException ex) {
78              throw new ModelException(ex);
79          }
80      }
81  
82      @Override
83      protected int doSize(@Nullable final Resource subj, @Nullable final URI pred,
84              @Nullable final Value obj, final Resource[] ctxs) {
85          try {
86              if (subj == null && pred == null && obj == null) {
87                  return (int) this.connection.size(ctxs);
88              } else {
89                  int size = 0;
90                  RepositoryResult<? extends Statement> iteration;
91                  iteration = this.connection.getStatements(subj, pred, obj, false, ctxs);
92                  try {
93                      while (iteration.hasNext()) {
94                          iteration.next();
95                          ++size;
96                      }
97                  } finally {
98                      iteration.close();
99                  }
100                 return size;
101             }
102         } catch (final RepositoryException ex) {
103             throw new ModelException(ex);
104         }
105     }
106 
107     @Override
108     protected Iterator<Statement> doIterator(@Nullable final Resource subj,
109             @Nullable final URI pred, @Nullable final Value obj, final Resource[] ctxs) {
110         try {
111             return new RepositoryIterator(this.connection.getStatements(subj, pred, obj, false,
112                     ctxs));
113         } catch (final RepositoryException ex) {
114             throw new ModelException(ex);
115         }
116     }
117 
118     @Override
119     protected boolean doAdd(@Nullable final Resource subj, @Nullable final URI pred,
120             @Nullable final Value obj, final Resource[] ctxs) {
121         try {
122             final long size = this.connection.size();
123             this.connection.add(subj, pred, obj, ctxs);
124             return this.connection.size() != size;
125         } catch (final RepositoryException ex) {
126             throw new ModelException(ex);
127         }
128     }
129 
130     @Override
131     protected boolean doRemove(@Nullable final Resource subj, @Nullable final URI pred,
132             @Nullable final Value obj, final Resource[] ctxs) {
133         try {
134             final long size = this.connection.size();
135             this.connection.remove(subj, pred, obj, ctxs);
136             return this.connection.size() != size;
137         } catch (final RepositoryException ex) {
138             throw new ModelException(ex);
139         }
140     }
141 
142     private static final class RepositoryIterator implements Iterator<Statement>, AutoCloseable {
143 
144         private final RepositoryResult<? extends Statement> iteration;
145 
146         RepositoryIterator(final RepositoryResult<? extends Statement> iteration) {
147             this.iteration = iteration;
148         }
149 
150         @Override
151         public boolean hasNext() {
152             try {
153                 return this.iteration.hasNext();
154             } catch (final RepositoryException ex) {
155                 close();
156                 throw new ModelException(ex);
157             }
158         }
159 
160         @Override
161         public Statement next() {
162             try {
163                 return this.iteration.next();
164             } catch (final RepositoryException ex) {
165                 close();
166                 throw new ModelException(ex);
167             }
168         }
169 
170         @Override
171         public void close() {
172             try {
173                 this.iteration.close();
174             } catch (final Throwable ex) {
175                 LOGGER.error("Could not close iteration", ex);
176             }
177         }
178 
179     }
