40  /**
41   * A specialized immutable {@code Namespace} set.
42   * <p>
43   * A {@code Namespaces} object is an immutable container of {@code prefix -> namespace URI}
44   * bindings, where multiple prefixes can map to the same namespace URI but there is at most a
45   * binding for any given prefix.
46   * </p>
47   * <p>
48   * Data in this class can be primarily accessed using lookup methods {@link #uriFor(String)},
49   * {@link #prefixFor(String)} and {@link #prefixesFor(String)}. In addition, the following views
50   * are offered:
51   * </p>
52   * <ul>
53   * <li>{@code set<Namespace object>} view, directly implemented by class instances;</li>
54   * <li>{@code prefix -> namespace URI} map view, produced by {@link #uriMap};</li>
55   * <li>{@code namespace URI -> prefix} map view, produced by {@link #prefixMap()};</li>
56   * <li>{@code set<prefix string>} view, produced by {@link #prefixes()};</li>
57   * <li>{@code set<URI string>} view, produced by {@link #uris()};</li>
58   * </ul>
59   * <p>
60   * Instances of this class can be created from:
61   * </p>
62   * <ul>
63   * <li>a {@code Namespace} iterable, by {@link #forIterable(Iterable, boolean)};</li>
64   * <li>a {@code prefix -> namespace URI} map, by {@link #forURIMap(Map)};</li>
65   * <li>a {@code namespace URI -> prefix} map, by {@link #forPrefixMap(Map, boolean)};</li>
66   * <li>a list of files identified by {@code URL}s, by {@link #load(Iterable, boolean)};</li>
67   * </ul>
68   * <p>
69   * This class is immutable and thus thread-safe.
70   * </p>
71   */
72  public final class Namespaces extends AbstractSet<Namespace> {
73

76      static {
77          try {
78              final List<URL> urls = new ArrayList<>();
79              final ClassLoader cl = Namespaces.class.getClassLoader();
80              for (final String p : new String[] { "META-INF/rdfpro.prefixes", "rdfpro.prefixes" }) {
81                  for (final Enumeration<URL> e = cl.getResources(p); e.hasMoreElements();) {
82                      urls.add(e.nextElement());
83                  }

85              DEFAULT = Namespaces.load(urls, true);
86          } catch (final IOException ex) {
87              throw new Error("Unexpected exception (!): " + ex.getMessage(), ex);
88          }

105     private Namespaces(final List<URIPrefixPair> pairs, final boolean resolveClashes) {
106 
107         final int size = pairs.size();
108         final int tableSize = size + size / 2; // target fill factor = 0.66
109         int neighborhood = 4; // first attempt
110 
111         this.uriMap = new EntryMap(true);
112         this.prefixMap = new EntryMap(false);
113         this.prefixTable = new int[2 * tableSize];
114         this.uriTable = new int[2 * tableSize];
115         this.data = new String[size * 2];
116 
117         List<URIPrefixPair> filteredPairs = pairs;
118         if (resolveClashes) {
119             final Set<String> prefixes = new HashSet<>();
120             filteredPairs = new ArrayList<URIPrefixPair>(pairs);
121             for (final Iterator<URIPrefixPair> i = filteredPairs.iterator(); i.hasNext();) {
122                 final URIPrefixPair pair = i.next();
123                 if (!prefixes.add(pair.prefix)) {
124                     i.remove(); // keep only first pair for given prefix
125                 }
126             }
127         }
128 
129         Collections.sort(pairs);
130 
131         int pointer = 0;
132         String uri = null;
133         int uriCount = 0;
134         for (int i = 0; i < size; ++i) {
135             final URIPrefixPair pair = pairs.get(i);
136             final String prefix = pair.prefix;
137             if (!pair.uri.equals(uri)) {
138                 uri = pair.uri;
139                 ++uriCount;
140             }
141             this.data[pointer++] = uri;
142             this.data[pointer] = prefix;
143             neighborhood = insert(this.uriTable, uri.hashCode(), pointer, neighborhood, null);
144             neighborhood = insert(this.prefixTable, prefix.hashCode(), pointer, neighborhood,
145                     prefix);
146             ++pointer;
147         }
148 
149         this.uriCount = uriCount;
150         this.neighborhood = neighborhood;
151     }
152 
153     private int insert(final int[] table, final int hash, final int pointer,
154             final int neighborhood, final String prefixToCheck) {
155 
156         // see http://en.wikipedia.org/wiki/Hopscotch_hashing
157 
158         final int buckets = table.length / 2;
159         int index = Math.abs(hash) % buckets;
160         int offset = index * 2;
161         int distance = 0;
162         while (table[offset] != 0) {
163             if (prefixToCheck != null && table[offset + 1] == hash
164                     && prefixToCheck.equals(this.data[table[offset]])) {
165                 throw new IllegalArgumentException("Duplicate prefix " + prefixToCheck);
166             }
167             index = (index + 1) % buckets;
168             offset = index * 2;
169             ++distance;
170         }
171         if (distance < neighborhood) {
172             table[offset] = pointer;
173             table[offset + 1] = hash;
174             return neighborhood;
175         }
176         for (int i = 1; i < neighborhood; ++i) {
177             int oldIndex = index - i;
178             oldIndex = oldIndex >= 0 ? oldIndex : oldIndex + buckets;
179             final int oldOffset = oldIndex * 2;
180             int oldDistance = index - Math.abs(table[oldOffset + 1]) % buckets;
181             oldDistance = oldDistance >= 0 ? oldDistance : oldDistance + buckets;
182             if (oldDistance < neighborhood) {
183                 table[offset] = table[oldOffset];
184                 table[offset + 1] = table[oldOffset + 1];
185                 table[oldOffset] = 0;
186                 table[oldOffset + 1] = 0;
187                 return insert(table, hash, pointer, neighborhood, prefixToCheck);
188             }
189         }
190         return insert(table, hash, pointer, neighborhood + 1, prefixToCheck);
191     }
192 
193     private int lookup(final String string, final int[] table, final int field) {
194         final int hash = string.hashCode();
195         int offset = 2 * (Math.abs(hash) % (table.length / 2));
196         for (int i = 0; i < this.neighborhood; ++i) {
197             final int prefixIndex = table[offset];
198             if (prefixIndex == 0) {
199                 return -1;
200             }
201             final int storedHash = table[offset + 1];
202             if (storedHash == hash) {
203                 final int stringIndex = prefixIndex - 1 + field;
204                 final String storedString = this.data[stringIndex];
205                 if (storedString.equals(string)) {
206                     return prefixIndex - 1;
207                 }
208             }
209             offset = (offset + 2) % table.length;
210         }
211         return -1;
212     }

214     /**
215      * Creates a {@code Namespaces} set by loading the namespace declarations in the files at the
216      * URLs specified. The text files must be encoded in UTF-8. Each line has the format
217      * {@code namespace_URI prefix_1 ... prefix_N}, where components are separated by whitespaces.
218      *
219      * @param urls
220      *            the URLs identifying the files to load, not null
221      * @param resolveClashes
222      *            true in case a clash caused by the same prefix being mapped to different
223      *            namespace URIs should be silently resolved by keeping only the first mapping in
224      *            the supplied iterable
225      * @return the created {@code Namespaces} object
226      * @throws IOException
227      *             on IO error
228      * @throws IllegalArgumentException
229      *             in case {@code resolveClashes} is false and the {@code Iterable} associates the
230      *             same prefix to different namespace URIs
231      */
232     public static Namespaces load(final Iterable<URL> urls, final boolean resolveClashes)
233             throws IOException {
234 
235         final List<URIPrefixPair> pairs = new ArrayList<>();
236 
237         for (final URL url : urls) {
238             try (final BufferedReader reader = new BufferedReader(new InputStreamReader(
239                     url.openStream(), Charset.forName("UTF-8")))) {
240                 String line;
241                 while ((line = reader.readLine()) != null) {
242                     final String[] tokens = line.split("\\s+");
243                     if (tokens.length >= 2 && !tokens[0].startsWith("#")) {
244                         for (int i = 1; i < tokens.length; ++i) {
245                             pairs.add(new URIPrefixPair(tokens[0], tokens[i]));
246                         }
247                     }
248                 }
249             }
250         }
251 
252         return new Namespaces(pairs, resolveClashes);
253     }

255     /**
256      * Returns a {@code Namespaces} set for the {@code Namespace} {@code Iterable} supplied. In
257      * case the {@code Iterable} is itself a {@code Namespaces} object, it will be directly
258      * returned.
259      *
260      * @param iterable
261      *            the {@code Iterable} where to load namespaces from, not null
262      * @param resolveClashes
263      *            true in case a clash caused by the same prefix being mapped to different
264      *            namespace URIs should be silently resolved by keeping only the first mapping in
265      *            the supplied iterable
266      * @return the corresponding {@code Namespaces} object
267      * @throws IllegalArgumentException
268      *             in case {@code resolveClashes} is false and the {@code Iterable} associates the
269      *             same prefix to different namespace URIs
270      */
271     public static Namespaces forIterable(final Iterable<? extends Namespace> iterable,
272             final boolean resolveClashes) throws IllegalArgumentException {
273         if (iterable instanceof Namespaces) {
274             return (Namespaces) iterable;
275         }
276         final int size = iterable instanceof Collection ? ((Collection<?>) iterable).size() : 1024;
277         final List<URIPrefixPair> pairs = new ArrayList<URIPrefixPair>(size);
278         for (final Namespace namespace : iterable) {
279             pairs.add(new URIPrefixPair(namespace.getName(), namespace.getPrefix()));
280         }
281         return new Namespaces(pairs, resolveClashes);
282     }

284     /**
285      * Returns a {@code Namespaces} set for the {@code prefix -> namespace URI} map supplied. In
286      * case the map is the {@link #uriMap()} of another {@code Namespaces} object, that object
287      * will be directly returned.
288      *
289      * @param map
290      *            the {@code prefix -> namespace URI} map, not null
291      * @return the corresponding {@code Namespaces} object
292      */
293     public static Namespaces forURIMap(final Map<String, String> map) {
294 
295         if (map instanceof EntryMap) {
296             final EntryMap entryMap = (EntryMap) map;
297             if (entryMap.keyIsPrefix) {
298                 return entryMap.getNamespaces();
299             }
300         }
301 
302         final List<URIPrefixPair> pairs = new ArrayList<URIPrefixPair>(map.size());
303         for (final Map.Entry<String, String> entry : map.entrySet()) {
304             pairs.add(new URIPrefixPair(entry.getValue(), entry.getKey()));
305         }
306         return new Namespaces(pairs, false); // no clashes possible
307     }
308 
309     /**
310      * Returns a {@code Namespaces} set for the {@code namespace URI -> URI} map supplied. In case
311      * the map is the {@link #prefixMap()} of another {@code Namespaces} object, that object will
312      * be directly returned.
313      *
314      * @param map
315      *            the {@code namespace URI -> prefix} map, not null
316      * @param resolveClashes
317      *            true in case a clash caused by the same prefix being mapped to different
318      *            namespace URIs should be silently resolved by keeping only the first mapping in
319      *            the supplied iterable
320      * @return the corresponding {@code Namespaces} object
321      * @throws IllegalArgumentException
322      *             in case {@code resolveClashes} is false and the {@code Iterable} associates the
323      *             same prefix to different namespace URIs
324      */
325     public static Namespaces forPrefixMap(final Map<String, String> map,
326             final boolean resolveClashes) throws IllegalArgumentException {
327 
328         if (map instanceof EntryMap) {
329             final EntryMap entryMap = (EntryMap) map;
330             if (!entryMap.keyIsPrefix) {
331                 return entryMap.getNamespaces();
332             }
333         }
334 
335         final List<URIPrefixPair> pairs = new ArrayList<URIPrefixPair>(map.size());
336         for (final Map.Entry<String, String> entry : map.entrySet()) {
337             pairs.add(new URIPrefixPair(entry.getKey(), entry.getValue()));
338         }
339         return new Namespaces(pairs, resolveClashes);
340     }

368     }

370     /**
371      * Returns all the prefixes bound to the namespace URI specified.
372      *
373      * @param uri
374      *            the namespace URI, not null
375      * @return a non-null list with the prefixes bound to the give URI
376      */
377     public List<String> prefixesFor(final String uri) {
378         int index = lookup(uri, this.uriTable, 0);
379         if (index < 0) {
380             return null;
381         }
382         final String storedURI = this.data[index];
383         final List<String> result = new ArrayList<>();
384         while (this.data[index] == storedURI && index < this.data.length) {
385             result.add(this.data[index + 1]);
386             index += 2;
387         }
388         return result;
389     }

391     /**
392      * Returns the first prefix bound to the namespace URI specified, or null if it does not
393      * exist.
394      *
395      * @param uri
396      *            the namespace URI, not null
397      * @return the first prefix bound to the namespace URI specified, if it exists, otherwise null
398      */
399     @Nullable
400     public String prefixFor(final String uri) {
401         final int index = lookup(uri, this.uriTable, 0);
402         return index < 0 ? null : this.data[index + 1];
403     }

456     private class EntryMap extends AbstractMap<String, String> {
457 
458         final boolean keyIsPrefix;
459 
460         final EntrySet entries;
461 
462         EntryMap(final boolean keyIsPrefix) {
463             this.keyIsPrefix = keyIsPrefix;
464             this.entries = new EntrySet(keyIsPrefix);
465         }
466 
467         Namespaces getNamespaces() {
468             return Namespaces.this;
469         }
470 
471         @Override
472         public Set<Map.Entry<String, String>> entrySet() {
473             return this.entries;
474         }
475 
476         @Override
477         public String get(final Object key) {
478             if (key instanceof String) {
479                 final String s = (String) key;
480                 return this.keyIsPrefix ? uriFor(s) : prefixFor(s);
481             }
482             return null;
483         }
484 
485         @Override
486         public boolean containsKey(final Object key) {
487             if (key instanceof String) {
488                 final String s = (String) key;
489                 return this.keyIsPrefix ? uriFor(s) != null : prefixFor(s) != null;
490             }
491             return false;
492         }
493 
494         @Override
495         public boolean containsValue(final Object value) {
496             if (value instanceof String) {
497                 final String s = (String) value;
498                 return this.keyIsPrefix ? prefixFor(s) != null : uriFor(s) != null;
499             }
500             return false;
501         }
502 
503     }

505     private class EntrySet extends AbstractSet<Map.Entry<String, String>> {
506 
507         final boolean keyIsPrefix;
508 
509         EntrySet(final boolean keyIsPrefix) {
510             this.keyIsPrefix = keyIsPrefix;
511         }
512 
513         @Override
514         public Iterator<Map.Entry<String, String>> iterator() {
515             return new EntryIterator(this.keyIsPrefix);
516         }
517 
518         @Override
519         public int size() {
520             return this.keyIsPrefix ? Namespaces.this.size() : Namespaces.this.uriCount;
521         }
522 
523         @Override
524         public boolean contains(final Object object) {
525             if (object instanceof Map.Entry<?, ?>) {
526                 final Map.Entry<?, ?> entry = (Map.Entry<?, ?>) object;
527                 if (entry.getKey() instanceof String && entry.getValue() instanceof String) {
528                     final String key = (String) entry.getKey();
529                     final String value = (String) entry.getValue();
530                     return this.keyIsPrefix ? Objects.equals(uriFor(key), value) : Objects.equals(
531                             prefixFor(key), value);
532                 }
533             }
534             return false;
535         }
536 
537     }

539     private class EntryIterator implements Iterator<Map.Entry<String, String>> {
540 
541         final boolean prefixToUri;
542 
543         int offset;
544 
545         EntryIterator(final boolean prefixToUri) {
546             this.prefixToUri = prefixToUri;
547             this.offset = 0;
548         }
549 
550         @Override
551         public boolean hasNext() {
552             return this.offset < Namespaces.this.data.length;
553         }
554 
555         @Override
556         public Entry<String, String> next() {
557             final String uri = Namespaces.this.data[this.offset];
558             final String prefix = Namespaces.this.data[this.offset + 1];
559             this.offset += 2;
560             if (this.prefixToUri) {
561                 return new AbstractMap.SimpleImmutableEntry<String, String>(prefix, uri);
562             } else {
563                 while (this.offset < Namespaces.this.data.length
564                         && Namespaces.this.data[this.offset] == Namespaces.this.data[this.offset - 2]) {
565                     this.offset += 2;
566                 }
567                 return new AbstractMap.SimpleImmutableEntry<String, String>(uri, prefix);
568             }
569         }
570 
571     }

573     private class NamespaceIterator implements Iterator<Namespace> {
574 
575         int offset;
576 
577         @Override
578         public boolean hasNext() {
579             return this.offset < Namespaces.this.data.length;
580         }
581 
582         @Override
583         public Namespace next() {
584             final String uri = Namespaces.this.data[this.offset];
585             final String prefix = Namespaces.this.data[this.offset + 1];
586             this.offset += 2;
587             return new NamespaceImpl(prefix, uri);
588         }

592     private static class URIPrefixPair implements Comparable<URIPrefixPair> {
593 
594         public final String uri;
595 
596         public final String prefix;
597 
598         public URIPrefixPair(final String uri, final String prefix) {
599             this.uri = Objects.requireNonNull(uri);
600             this.prefix = Objects.requireNonNull(prefix);
601         }
602 
603         @Override
604         public int compareTo(final URIPrefixPair other) {
605             int result = this.uri.compareTo(other.uri);
606             if (result == 0) {
607                 result = this.prefix.compareTo(other.prefix);
608             }
609             return result;
610         }
611 
612         @Override
613         public boolean equals(final Object object) {
614             if (object == this) {
615                 return true;
616             }
617             if (!(object instanceof URIPrefixPair)) {
618                 return false;
619             }
620             final URIPrefixPair other = (URIPrefixPair) object;
621             return this.uri.equals(other.uri) && this.prefix.equals(other.prefix);
622         }
623 
624         @Override
625         public int hashCode() {
626             return this.uri.hashCode() ^ this.prefix.hashCode();
627         }
